def analyze_legal_dataset(df):
    """
    Ανάλυση νομικών υποθέσεων με καλύτερη οπτικοποίηση
    """
    # 1. Case Outcome Distribution
    plt.figure(figsize=(15, 5))
    
    # Pie chart των αποτελεσμάτων
    plt.subplot(131)
    outcome_counts = df['case_outcome'].value_counts()
    plt.pie(outcome_counts.values, labels=outcome_counts.index, autopct='%1.1f%%')
    plt.title('Κατανομή Αποτελεσμάτων Υποθέσεων')
    
    # Year Distribution
    plt.subplot(132)
    # Εξαγωγή ετών από το case_text
    years = df['case_text'].str.extract(r'(\d{4})').astype(int)
    plt.hist(years, bins=20)
    plt.title('Κατανομή Ετών Υποθέσεων')
    plt.xlabel('Έτος')
    plt.ylabel('Αριθμός Υποθέσεων')
    
    # Citation Analysis
    plt.subplot(133)
    citations_per_case = df['case_text'].str.count('v.')
    plt.hist(citations_per_case, bins=range(min(citations_per_case), max(citations_per_case) + 2, 1))
    plt.title('Αναφορές ανά Υπόθεση')
    plt.xlabel('Αριθμός Αναφορών')
    plt.ylabel('Αριθμός Υποθέσεων')
    
    plt.tight_layout()
    plt.show()
    
    # Print Summary Statistics
    print("\n📊 Στατιστικά Ανάλυσης:")
    print(f"Συνολικές υποθέσεις: {len(df):,}")
    print("\nΚατανομή αποτελεσμάτων:")
    for outcome, count in outcome_counts.items():
        percentage = (count/len(df)) * 100
        print(f"- {outcome}: {count:,} ({percentage:.1f}%)")
    
    print("\nΧρονική Ανάλυση:")
    print(f"- Παλαιότερη υπόθεση: {years.min()[0]}")
    print(f"- Πιο πρόσφατη υπόθεση: {years.max()[0]}")
    print(f"- Διάμεσο έτος: {years.median()[0]:.0f}")
    
    print("\nΑνάλυση Αναφορών:")
    print(f"- Μέσος αριθμός αναφορών: {citations_per_case.mean():.1f}")
    print(f"- Ελάχιστες αναφορές: {citations_per_case.min()}")
    print(f"- Μέγιστες αναφορές: {citations_per_case.max()}")

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime

# %%
def analyze_temporal_patterns(df):
    """
    Ανάλυση χρονικών patterns με έμφαση στη σχέση
    αποφάσεων και χρονικής περιόδου
    """
    # Εξαγωγή έτους από το κείμενο
    df['year'] = df['case_text'].str.extract(r'(\d{4})').astype(int)
    
    # 1. Ανάλυση τάσεων ανά πενταετία
    df['period'] = ((df['year'] - 1990) // 5) * 5 + 1990
    
    # Δημιουργία visualizations
    fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(15, 12))
    
    # 1. Εξέλιξη αποφάσεων στο χρόνο
    outcomes_by_year = pd.crosstab(df['year'], df['case_outcome'])
    outcomes_by_year.plot(kind='bar', stacked=True, ax=ax1)
    ax1.set_title('Εξέλιξη Αποφάσεων ανά Έτος')
    ax1.set_xlabel('Έτος')
    ax1.set_ylabel('Αριθμός Υποθέσεων')
    plt.setp(ax1.xaxis.get_majorticklabels(), rotation=45)
    
    # 2. Ποσοστιαία κατανομή ανά περίοδο
    outcomes_by_period = pd.crosstab(df['period'], df['case_outcome'], normalize='index') * 100
    outcomes_by_period.plot(kind='area', ax=ax2)
    ax2.set_title('Ποσοστιαία Κατανομή ανά Πενταετία')
    ax2.set_xlabel('Περίοδος')
    ax2.set_ylabel('Ποσοστό (%)')
    
    # 3. Success Rate (Allowed) ανά περίοδο
    success_rate = df[df['case_outcome'] == 'Allowed'].groupby('period').size() / \
                  df.groupby('period').size() * 100
    success_rate.plot(marker='o', ax=ax3)
    ax3.set_title('Success Rate ανά Πενταετία')
    ax3.set_xlabel('Περίοδος')
    ax3.set_ylabel('Ποσοστό Επιτυχίας (%)')
    
    # 4. Settlement Rate trends
    settlement_rate = df[df['case_outcome'] == 'Settled'].groupby('period').size() / \
                     df.groupby('period').size() * 100
    settlement_rate.plot(marker='o', ax=ax4)
    ax4.set_title('Settlement Rate ανά Πενταετία')
    ax4.set_xlabel('Περίοδος')
    ax4.set_ylabel('Ποσοστό Διακανονισμών (%)')
    
    plt.tight_layout()
    plt.show()
    
    # Εκτύπωση στατιστικών
    print("\n📊 Χρονική Ανάλυση Αποφάσεων:")
    
    # Ανάλυση trends
    recent_period = df[df['year'] >= 2010]
    early_period = df[df['year'] < 2000]
    
    print("\nΣύγκριση Περιόδων:")
    for outcome in ['Allowed', 'Dismissed', 'Settled']:
        recent_rate = (recent_period['case_outcome'] == outcome).mean() * 100
        early_rate = (early_period['case_outcome'] == outcome).mean() * 100
        print(f"\n{outcome}:")
        print(f"- Πρόσφατη περίοδος (2010+): {recent_rate:.1f}%")
        print(f"- Παλαιότερη περίοδος (<2000): {early_rate:.1f}%")
        print(f"- Μεταβολή: {recent_rate - early_rate:+.1f}%")

# %%
if df is not None:
    analyze_temporal_patterns(df)
